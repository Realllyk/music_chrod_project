# Backend runtime dependencies for Windows + Conda.
#
# Recommended setup:
#   conda create -n music_chord_backend python=3.10 -y
#   conda activate music_chord_backend
#   python -m pip install --upgrade pip
#   pip install -r requirements_windows_conda.txt
#
# Scope:
# - Flask backend APIs
# - MySQL access through PyMySQL
# - Aliyun OSS uploads used by artist/audio-source endpoints
# - Default librosa-based melody/chord transcription configured in config.json
#
# Not included:
# - spleeter / tensorflow
# - demucs / torch
#
# Install those separately only if backend/config.json switches
# transcription.algorithm.melody or transcription.algorithm.chord away from
# "librosa".
